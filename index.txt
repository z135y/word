<script>
    function sync() {
        const rawText = document.getElementById('textInput').value;
        const container = document.getElementById('main-container');
        document.getElementById('counter').innerText = `字數統計：${rawText.replace(/\s/g, '').length} 字`;

        // 處理格式：自動偵測換行並縮進兩格
        const lines = rawText.split('\n');
        let processedText = "";
        lines.forEach(line => {
            const trimmed = line.trim();
            if (trimmed.length > 0) {
                processedText += "　　" + trimmed;
            }
        });

        container.innerHTML = '';
        const rowsPerCol = 20; 
        const colsPerPage = 15; 
        const totalCells = rowsPerCol * colsPerPage; // 300字一頁

        if (processedText.length === 0) {
            renderPage("", 1);
            return;
        }

        // 按照 300 字一頁進行切割
        for (let i = 0; i < processedText.length; i += totalCells) {
            const chunk = processedText.substring(i, i + totalCells);
            renderPage(chunk, Math.floor(i / totalCells) + 1);
        }

        function renderPage(content, pageNum) {
            const block = document.createElement('div');
            block.className = 'page-block';

            // --- 示範區 (Top) ---
            const label1 = document.createElement('div');
            label1.className = 'area-label';
            label1.innerText = `示範區 (第 ${pageNum} 頁)`;
            block.appendChild(label1);

            const grid1 = document.createElement('div');
            grid1.className = 'manuscript-grid';
            for (let j = 0; j < totalCells; j++) {
                const cell = document.createElement('div');
                cell.className = 'cell';
                cell.innerText = content[j] || '';
                grid1.appendChild(cell);
            }
            block.appendChild(grid1);

            block.appendChild(document.createElement('div')).className = 'divider';

            // --- 抄寫區 (Bottom) ---
            const label2 = document.createElement('div');
            label2.className = 'area-label';
            label2.innerText = `練習抄寫區`;
            block.appendChild(label2);

            const grid2 = document.createElement('div');
            grid2.className = 'manuscript-grid';
            for (let j = 0; j < totalCells; j++) {
                const cell = document.createElement('div');
                cell.className = 'cell';
                cell.innerText = ''; 
                grid2.appendChild(cell);
            }
            block.appendChild(grid2);

            container.appendChild(block);
        }
    }

    window.onload = sync;
</script>
